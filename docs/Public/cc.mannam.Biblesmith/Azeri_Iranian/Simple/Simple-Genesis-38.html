او زامان يهودا قارداشلاريندان آيريلاراق خئرا آدلي عَدولّاملي بئر آدامين يانينا گتدي. .1
يهودا اورادا شوعه آدلي کَنَعانلي بئر آدامين قيزيني گؤردو. يهودا قيزي آرواد آليب، اونونلا ياتدي. .2
آرواد بويلو اولوب بئر اوغلان دوغدو و يهودا اونون آديني عِر قويدو. .3
آرواد گئنه بويلو اولوب بئر اوغلان دوغدو و آديني اونان قويدو. .4
سونرا گئنه ده بئر اوغلان دوغوب، آديني شِلا قويدو. آرواد اونو دوغدوغو زامان، يهودا کِزئبده ائدي. .5
يهودا ائلک اوغلو عِر اوچون بئر آرواد آلدي، اونون آدي تامار ائدي. .6
يهودانين ائلک اوغلو عِر ربّئن گؤزونده پئس ائدي و رب اونو اؤلدوردو. .7
يهودا اونانا ددي: "قارداشين آروادينلا يات، اؤز قايينليق وظئفه‌ني يرئنه يتئرئب قارداشين اوچون نسئل يتئشدئر." .8
اونان بئلئردي کي، بو نسئل اونون اؤزونون اولماياجاق. بونا گؤره ده قارداشي آروادي ائله ياخينليق ادنده توخومونو يره تؤکدو کي، قارداشي اوچون نسئل يتئشدئرمه‌سئن. .9
اونون اتدئيي بو ائش ربّئن گؤزونده پئس اولدو و رب اونو دا اؤلدوردو. .10
يهودا گلئني تامارا ددي: "اوغلوم شِلا بؤيوينه قدر آتانين اِوئنده دول اولاراق قال." يهودا فئکئرلشدي کي، شِلا دا قارداشلاري کئمي اؤله بئلر. تامار گدئب آتاسينين اِوئنده قالدي. .11
اوزون مودّتدن سونرا شوعه‌نئن قيزي اولان يهودانين آروادي اؤلدو. يهودا ماتم قورتاراندان سونرا، دوستو اَدوللاملي خئره ائله بئرلئکده تئمه‌ناعيا، سوروسونو قيرخانلارين يانينا گتدي. .12
تامارا بئلدئردئلر: "باخ قايناتان سوروسونو قيرخماق اوچون تئمه‌ناعيا گلئر." .13
تامار دوللوق پالتاريني اَينئندن چيخارديب روبند اؤرتدو و بورونوب، تئمه‌ناعيا گدن يولدا اولان عِنايئم دروازاسينين يانيندا اوتوردو. چونکي او گؤروردو کي، شِلا بؤيويوب، لاکئن او حله ده شِلايا اَره ورئلمه‌يئب. .14
يهودا اونو گؤرنده اله بئلدي کي، بئر فاحئشه آرواددير، چونکي اوزونو اؤرتموشدو. .15
يهودا يولونو اونا طرف ساليب ددي: "لوطفن ائجازه ور، سنئنله ياتيم." آمّا اونون اؤز گلئني اولدوغونو بئلمه‌دي. آرواد ددي: "منئمله ياتماق اوچون منه نه ورئرسن؟" .16
يهودا ددي: "سورودن بئر کچي بالاسي گؤندره‌رم." او ددي: "اونو گؤندرَنه قدر منه گئروو اولاراق بئر شي ورَرسن؟" .17
يهودا سوروشدو: "گئروو اولاراق سنه نه ورئم؟" آرواد ددي: "اؤز نو، قايتانيني و اَلئنده اولان عصاني." يهودا بونلاري آروادا ورئب، اونونلا ياتدي. آرواد اوندان بويلو اولدو. .18
سونرا تامار قالخيب گتدي و روبندئني گؤتوروب دوللوق پالتاريني گِيدي. .19
يهودا آروادين اَلئندن گئرووونو آلماق اوچون اَدولّلاملي دوستونون واسئطه‌سئله کچي بالاسيني گؤندردي. آمّا او، آروادي تاپمادي. .20
آروادين اولدوغو يرئن آداملاريندان سوروشدو: "عِنايئمده يول قيراغيندا اولان معبد فاحئشه‌سي هارادادير؟" اونلار ددئلر: "بورادا معبد فاحئشه‌سي يوخدور." .21
او، يهودانين يانينا قاييديب ددي: "اونو تاپا بئلمه‌دئم. بوندان علاوه او يرئن آداملاري دا ددئلر کي، "بورادا معبد فاحئشه‌سي يوخدور." .22
يهودا ددي: "قوي بو شيلر اونونکو اولسون، مبادا بئز بو ائشدن روسواي اولاق. باخ من بو کچي بالاسيني اونا گؤندردئم، آمّا سن اونو تاپمادين." .23
بوندان تخمئنن اوچ آي سونرا يهودايا بئلدئردئلر: "گلئنئن تامار فاحئشه‌لئک ادئب، هم ده بو فاحئشه‌لئکدن بويلو اولوب." يهودا ددي: "اونو چيخارديب يانديرين." .24
او اِشئيه چيخاريلديغي زامان قايناتاسينا خبر گؤندرئب ددي: "من بو شيلرئن صاحئبي اولان آدامدان بويلو اولدوم." سونرا ددي: "خاهئش ادئرم، باخ گؤر، بو مؤهور، قايتان و عصا کئمئندئر؟" .25
يهودا بونلاري تانييب ددي: "او مندن حاقليدير، چونکي اونو اوغلوم شِلايا ورمه‌دئم." يهودا اونونلا بئر داها ياخينليق اتمه‌دي. .26
تامار دوغاندا معلوم اولدو کي، بطنئنده اِکئز وار. .27
دوغوش واختيندا بئر اوشاق اَلئني چيخارتدي. ماما گؤتوروب اونون اَلئنه قيرميزي ائپ باغلادي و ددي: "اوّلجه بو چيخدي." .28
آمّا او اوشاق اَلئني گري چکدي و قارداشي بئرئنجي چيخدي. ماما ددي: "اؤزونه نجه دِشئک آچدين؟" بونا گؤره ده اونون آديني پِرِص قويدولار. .29
اوندان سونرا اَلئنده قيرميزي ائپ اولان قارداشي دوغولدو. اونون آديني زِرَخ قويدولار. .30
